];

let categories = ["All", "Kurtas", "Sarees", "Lehengas", "Sherwani", "Anarkali"];
let cart = [];

function loadFromStorage() {
    const savedProducts = localStorage.getItem('desiProducts');
    if (savedProducts) products = JSON.parse(savedProducts);
    
    const savedCategories = localStorage.getItem('desiCategories');
    if (savedCategories) categories = JSON.parse(savedCategories);
    
    const savedBG = localStorage.getItem('heroBackground');
    if (savedBG) document.getElementById('hero').style.backgroundImage = `url(${savedBG})`;
}

function saveToStorage() {
    localStorage.setItem('desiProducts', JSON.stringify(products));
    localStorage.setItem('desiCategories', JSON.stringify(categories));
}

function renderCategories() {
    const container = document.getElementById('existing-categories');
    container.innerHTML = '';
    categories.forEach(cat => {
        const btn = document.createElement('button');
        btn.className = `category-tab ${cat === 'All' ? 'active' : ''}`;
        btn.textContent = cat;
        btn.onclick = () => filterByCategory(cat);
        container.appendChild(btn);
    });
}

function filterByCategory(cat) {
    const filtered = cat === 'All' ? products : products.filter(p => p.category === cat);
    renderProducts(filtered);
    document.querySelectorAll('.category-tab').forEach(tab => tab.classList.toggle('active', tab.textContent === cat));
}

function renderProducts(filteredProducts) {
    const grid = document.getElementById('products-grid');
    grid.innerHTML = '';
    filteredProducts.forEach(product => {
        const card = document.createElement('div');
        card.className = 'product-card';
        card.innerHTML = `
            <img src="\( {product.img}" alt=" \){product.name}">
            <div class="product-info">
                <h3>${product.name}</h3>
                <a href="${product.visitLink}" target="_blank" class="visit-button">Visit Product Link</a>
                <p class="price">₹${product.price}</p>
                <button class="add-to-cart" onclick="addToCart(${product.id})">Add to Cart</button>
            </div>
        `;
        grid.appendChild(card);
    });
}

function addToCart(id) {
    const prod = products.find(p => p.id === id);
    if (prod) {
        cart.push(prod);
        document.getElementById('cart-count').textContent = cart.length;
        alert(prod.name + " added to cart! 🛒");
    }
}

document.getElementById('create-category-form').addEventListener('submit', function(e) {
    e.preventDefault();
    const newCat = document.getElementById('new-category-name').value.trim();
    if (newCat && !categories.includes(newCat)) {
        categories.push(newCat);
        saveToStorage();
        renderCategories();
        populateCategorySelect();
        alert(`New Folder "${newCat}" created successfully!`);
        this.reset();
    }
});

function populateCategorySelect() {
    const select = document.getElementById('new-category');
    select.innerHTML = '<option value="">Select Folder/Category</option>';
    categories.forEach(cat => {
        if (cat !== 'All') {
            const option = document.createElement('option');
            option.value = cat;
            option.textContent = cat;
            select.appendChild(option);
        }
    });
}

document.getElementById('add-product-form').addEventListener('submit', function(e) {
    e.preventDefault();
    const newProd = {
        id: Date.now(),
        name: document.getElementById('new-name').value,
        price: parseInt(document.getElementById('new-price').value),
        img: document.getElementById('new-img').value || 'https://picsum.photos/id/999/600/600',
        visitLink: document.getElementById('new-visit-link').value,
        category: document.getElementById('new-category').value
    };
    products.unshift(newProd);
    saveToStorage();
    renderProducts(products);
    alert('✅ Product published successfully!');
    this.reset();
});

document.getElementById('bg-upload').addEventListener('change', function(e) {
    const file = e.target.files[0];
    if (file) {
        const reader = new FileReader();
        reader.onload = ev => {
            document.getElementById('hero').style.backgroundImage = `url(${ev.target.result})`;
            localStorage.setItem('heroBackground', ev.target.result);
        };
        reader.readAsDataURL(file);
    }
});

document.getElementById('logo-upload').addEventListener('change', function(e) {
    const file = e.target.files[0];
    if (file) {
        const reader = new FileReader();
        reader.onload = ev => document.getElementById('site-logo').src = ev.target.result;
        reader.readAsDataURL(file);
    }
});

document.getElementById('search-btn').addEventListener('click', () => {
    const query = document.getElementById('search-input').value.toLowerCase().trim();
    const filtered = products.filter(p => p.name.toLowerCase().includes(query));
    renderProducts(filtered.length ? filtered : products);
});
